#include <LiquidCrystal_I2C.h>
#include <Wire.h>
LiquidCrystal_I2C lcd(0x27,20,4);

void setup(){
  lcd.init();
  lcd.backlight();
  lcd.home();
  Serial.begin(9600);
}
void loop(){
  if(Serial.available()>0){
    String receive = Serial.readString();
    receive.trim();
    lcd.clear();
    lcd.print(receive);
    Serial.println(receive);
  }
}
